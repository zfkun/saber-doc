define(function () {
    /**
     * HTML编码
     *
     * @public
     * @param {string} str
     * @return {string}
     */
    function encodeHTML(str) {
        if (!str) {
            return '';
        }

        // 只需将可能与HTML产生冲突的关键字符encode即可
        return String(str).replace(/&/g, '&')
                .replace(/</g, '<')
                .replace(/>/g, '>')
                .replace(/"/g, '"')
                .replace(/'/g, ''');
    }

    return encodeHTML;
});
